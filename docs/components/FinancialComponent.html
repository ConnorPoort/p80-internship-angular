  getBitData() {
    setTimeout(() => {
      this.getBitData()
    }, 10000); // time it calls
    this.http.get('https://api.coinmarketcap.com/v1/ticker/?start=0&limit=24&convert=EUR').subscribe(
      res=>{ this.bitVal = res
      }
    )
  }
